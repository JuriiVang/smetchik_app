import 'package:flutter/material.dart';
import 'package:cloud_firestore/cloud_firestore.dart';

class BuildingListScreen extends StatelessWidget {
  final String projectId;
  const BuildingListScreen({super.key, required this.projectId});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('ðŸ¢ Ð¡Ð¿Ð¸ÑÐ¾Ðº Ð·Ð´Ð°Ð½Ð¸Ð¹')),
      body: StreamBuilder(
        stream: FirebaseFirestore.instance
            .collection('projects')
            .doc(projectId)
            .collection('buildings')
            .snapshots(),
        builder: (context, snapshot) {
          if (!snapshot.hasData) {
            return const Center(child: CircularProgressIndicator());
          }

          var buildings = snapshot.data!.docs;

          return ListView.builder(
            itemCount: buildings.length,
            itemBuilder: (context, index) {
              var building = buildings[index];
              return ListTile(
                title: Text(building['name']),
                subtitle: Text("Ð­Ñ‚Ð°Ð¶ÐµÐ¹: ${building['floors']}"),
                trailing: const Icon(Icons.arrow_forward_ios),
                onTap: () {
                  Navigator.push(
                    context,
                    MaterialPageRoute(
                      builder: (context) => RoomListScreen(
                        projectId: projectId,
                        buildingId: building.id,
                      ),
                    ),
                  );
                },
              );
            },
          );
        },
      ),
    );
  }
}
